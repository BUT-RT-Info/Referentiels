% Contenus
\ajoutRcontenudetaille{
Objectifs visés
\begin{itemize}
\item
  Développer sa confiance en soi
\item
  Se présenter, présenter quelqu'un, interroger
\item
  Savoir structurer son discours oral et écrit (courriel, conversation
  téléphonique, visioconférence\ldots)
\item
  Décrire, expliquer un élément technique
\item
  Savoir écouter, comprendre et analyser les demandes de son
  interlocuteur, suggérer des solutions
\item
  Reformuler, expliciter un message
\item
  Appréhender le vocabulaire technique des domaines cibles
\item
  Extensions possibles : télécollaboration, télétandem.
\end{itemize}
}
